    <div class="container">
        <h1>Marks Calculator</h1>
        <button onclick="getStudentMarks()">Enter Marks</button>
        <div id="results"></div>
    </div>
    <script>
        class Student {
            constructor(name, subject1, subject2, subject3) {
                this.name = name;
                this.subject1 = subject1;
                this.subject2 = subject2;
                this.subject3 = subject3;
            }

            calculateTotal() {
                return this.subject1 + this.subject2 + this.subject3;
            }

            calculateAverage() {
                return this.calculateTotal() / 3;
            }
        }

        function getStudentMarks() {
            const numberOfStudents = parseInt(prompt("Enter the number of students:"), 10);
            const students = [];

            for (let i = 0; i < numberOfStudents; i++) {
                const name = prompt(`Enter the name of Student ${i + 1}:`);
                const subject1 = parseFloat(prompt(`Enter marks for ${name} - Subject 1:`));
                const subject2 = parseFloat(prompt(`Enter marks for ${name} - Subject 2:`));
                const subject3 = parseFloat(prompt(`Enter marks for ${name} - Subject 3:`));

                const student = new Student(name, subject1, subject2, subject3);
                students.push(student);
            }

            displayResults(students);
        }

        function displayResults(students) {
            const resultsDiv = document.getElementById('results');
            resultsDiv.innerHTML = '';

            for (let i = 0; i < students.length; i++) {
                const total = students[i].calculateTotal();
                const average = students[i].calculateAverage();

                const studentDiv = document.createElement('div');
                studentDiv.className = 'student-result';
                studentDiv.innerHTML = `<p><strong>${students[i].name}</strong></p>
                                        <p>Total Marks: ${total}</p>
                                        <p>Average Marks: ${average.toFixed(2)}</p>`;

                resultsDiv.appendChild(studentDiv);
            }
        }
    </script>
